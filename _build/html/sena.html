dt2 = res
dt2['Rango_edad'] = dt2.apply(lambda row: [[[['E','D'][row.edad < 50], 'C'][row.edad < 40], 'B'][row.edad < 30], 'A'][row.edad < 20], axis = 1)

fig = sns.relplot(x='Rango_edad' ,y='valor_seguro',hue='imc',data=dt2)
